\section*{第2章}

\begin{itemize}
  \item 22ページの $b_1 \land b_2$ のための操作的意味論の3つ目の規則が
    \infrule
    {\tuple{b_0, \sigma} \arrow \Ttrue \andalso \tuple{b_1, \sigma} \arrow \Ttrue}
    {\tuple{b_0 \land b_1, \sigma} \arrow \Tfalse}
    とあるのは
    \infrule
    {\tuple{b_0, \sigma} \arrow \Ttrue \andalso \tuple{b_1, \sigma} \arrow \Ttrue}
    {\tuple{b_0 \land b_1, \sigma} \arrow \Ttrue}
    の誤りです．
\end{itemize}
